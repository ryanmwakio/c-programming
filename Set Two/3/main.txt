{

    char gender = "m";

    // Input
    printf("\nEnter Gender(m/f): ");

    scanf("%d", &earning);

    // Computation
    // female
    if ((gender!="m") && (earning<15000))

        printf("\n\tYour tax rate is 14 percent.\n");

    // male
    if ((gender=="m") && (earning<14000))
        printf("\n\tYour tax rate is 13 percent.\n");
